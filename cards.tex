% #1 = color; #2 = value; #3 = suit; #4 = image
\newcommand\playingcard[4]{%
  % Outer box of the card.
  \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,colframe=black,left=0pt,right=1pt,top=0pt,bottom=0pt]%

    % (first two)(third) so that I can bottom align them
    \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,frame empty,
             left=0pt,right=0pt,top=0pt,bottom=0pt,box align=bottom]%

       % First two elements -- top aligned
       \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,frame empty,
                         left=0pt,right=0pt,top=0pt,bottom=0pt,box align=top]%
         \begin{tabular}{@{}c@{}} {\color{#1}\Large #2}\\ {\color{#1}\Large $#3$} \end{tabular}%
       \end{tcolorbox}%

       \begin{tcolorbox}[hbox,nobeforeafter, boxsep=0pt, frame empty,
                left=0pt,right=0pt,top=\baselineskip,bottom=\baselineskip,box align=top]%
       \begin{tcolorbox}[nobeforeafter, boxsep=0pt, sharp corners,boxrule=0.5pt, colback={bisque},
           left=0pt,right=0pt,top=0pt,bottom=0pt,box align=top,width=0.75in,height=1.25in,halign=center,
           valign=center]%
              {\color{#1}#4}%
           \end{tcolorbox}%
           \end{tcolorbox}%

    \end{tcolorbox}%

    % Third box -- bottom aligned to the first two
    \begin{tcolorbox}[hbox,nobeforeafter, boxsep=0pt,frame empty,
        left=0pt,right=0pt,top=0pt,bottom=0pt,box align=bottom]%
      \rotatebox{180}{\begin{tabular}{@{}c@{}}{\color{#1}\Large #2}\\\Large{\color{#1}$#3$}\end{tabular}}%
      \end{tcolorbox}%
  \end{tcolorbox}}%

\newcommand\royalcard[4]{%
  % Outer box of the card.
  \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,colframe=black,left=0pt,right=1pt,top=0pt,bottom=0pt]%

    % (first two)(third) so that I can bottom align them
    \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,frame empty,
             left=0pt,right=0pt,top=0pt,bottom=0pt,box align=bottom]%

       % First two elements -- top aligned
       \begin{tcolorbox}[hbox,nobeforeafter,boxsep=0pt,frame empty,
                         left=0pt,right=0pt,top=0pt,bottom=0pt,box align=top]%
         \begin{tabular}{@{}c@{}} {\color{#1}\Large #2}\\ {\color{#1}\Large $#3$} \end{tabular}%
       \end{tcolorbox}%

       \begin{tcolorbox}[hbox,nobeforeafter, boxsep=0pt, frame empty,
                left=0pt,right=0pt,top=\baselineskip,bottom=\baselineskip,box align=top]%
       \begin{tcolorbox}[nobeforeafter, boxsep=0pt, sharp corners,boxrule=0.5pt,width=0.75in,height=1.25in,halign=center,valign=center,colback={bisque},
                left=0pt,right=0pt,top=0pt,bottom=0pt,box align=top]%
              \includesvg[width=0.75in]{#4}%
           \end{tcolorbox}%
           \end{tcolorbox}%

    \end{tcolorbox}%

    % Third box -- bottom aligned to the first two
    \begin{tcolorbox}[hbox,nobeforeafter, boxsep=0pt,frame empty,
        left=0pt,right=0pt,top=0pt,bottom=0pt,box align=bottom]%
      \rotatebox{180}{\begin{tabular}{@{}c@{}}{\color{#1}\Large #2}\\\Large{\color{#1}$#3$}\end{tabular}}%
      \end{tcolorbox}%
  \end{tcolorbox}}%
